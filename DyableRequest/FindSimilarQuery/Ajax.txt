<script type="text/javascript">

function loadXMLDoc(doc_id, node_id, excerpt_id, query, focus, word_id_set)
{
var xmlhttp; 
if (window.XMLHttpRequest)
  {// code for IE7+, Firefox, Chrome, Opera, Safari
  xmlhttp=new XMLHttpRequest();
  }
else
  {// code for IE6, IE5
  xmlhttp=new ActiveXObject("Microsoft.XMLHTTP");
  }
xmlhttp.onreadystatechange=function()
  {
  if (xmlhttp.readyState==4 && xmlhttp.status==200)
    {
    
        var text = xmlhttp.responseText;
        

	var offset = 0;
        for(var i=0; i<text.length; i++) {
            if(text.charAt(i) == '`') {
                offset = i;
                break;
            }
        }


        var summary_tag = "summary";
        summary_tag += excerpt_id.toString();
        document.getElementById(summary_tag).innerHTML = "<font size=\"0\">" + text.substring(0, offset) + "</font>";
        document.getElementById(summary_tag).style.display="block";
        
        var excerpt_tag = "excerpt";
        excerpt_tag += excerpt_id.toString();
        document.getElementById(excerpt_tag).innerHTML = doc_id.toString() + " " + text.substring(offset+1, text.length);
        document.getElementById(excerpt_tag).style.display="none";
    }
  }

var text = "http://localhost/cgi-bin/DocumentQuery?doc=";
text += doc_id.toString();
text += "&q=";
text += query;
text += "&f=";
text += focus;
text += "&f=";
text += "&node=";
text += node_id;
text += "&id_set=";
text += word_id_set;


xmlhttp.open("GET", text, true);
xmlhttp.send();
}
</script>
